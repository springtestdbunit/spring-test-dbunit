16  package com.github.springtestdbunit.annotation;
17  
18  /**
19   * Database test operations that can be performed to configure database tables.
20   * 
21   * @see DatabaseSetup
22   * @see DatabaseTearDown
23   * 
24   * @author Phillip Webb
25   */
26  public enum DatabaseOperation {
27  
28  	/**
29  	 * Updates the contents of existing database tables from the dataset.
30  	 */
31  	UPDATE,
32  
33  	/**
34  	 * Inserts new database tables and contents from the dataset.
35  	 */
36  	INSERT,
37  
38  	/**
39  	 * Refresh the contents of existing database tables. Rows from the dataset will insert or replace existing data. Any
40  	 * database rows that are not in the dataset remain unaffected.
41  	 */
42  	REFRESH,
43  
44  	/**
45  	 * Deletes database table rows that matches rows from the dataset.
46  	 */
47  	DELETE,
48  
49  	/**
50  	 * Deletes all rows from a database table when the table is specified in the dataset. Tables in the database but not
51  	 * in the dataset remain unaffected.
52  	 * @see #TRUNCATE_TABLE
53  	 */
54  	DELETE_ALL,
55  
56  	/**
57  	 * Deletes all rows from a database table when the table is specified in the dataset. Tables in the database but not
58  	 * in the dataset are unaffected. Identical to {@link #DELETE_ALL} expect this operation cannot be rolled back and
59  	 * is supported by less database vendors.
60  	 * @see #DELETE_ALL
61  	 */
62  	TRUNCATE_TABLE,
63  
64  	/**
65  	 * Deletes all rows from a database table when the tables is specified in the dataset and subsequently insert new
66  	 * contents. Equivalent to calling {@link #DELETE_ALL} followed by {@link #INSERT}.
67  	 */
68  	CLEAN_INSERT;
69  
70  }
