33  /**
34   * Test annotation that can be used to assert that a database is in given state after tests have run.
35   *
36   * @see DbUnitTestExecutionListener
37   *
38   * @author Phillip Webb
39   * @author Mario Zagar
40   */
41  @Documented
42  @Inherited
43  @Retention(RetentionPolicy.RUNTIME)
44  @Target({ ElementType.TYPE, ElementType.METHOD })
45  @Repeatable(ExpectedDatabases.class)
46  public @interface ExpectedDatabase {
47  
48  	/**
49  	 * The name of the connection that should be used when verifying data. Can refer to a connection specified in
50  	 * {@link DbUnitConfiguration @DbUnitConfiguration} or left blank to use the default connection.
51  	 * @return the connection
52  	 */
53  	String connection() default "";
54  
55  	/**
56  	 * Provides the location of the dataset that will be used to test the database.
57  	 * @return The dataset locations
58  	 * @see DbUnitConfiguration#dataSetLoader()
59  	 */
60  	String value() default "";
61  
62  	/**
63  	 * Database assertion mode to use. Default is {@link DatabaseAssertionMode#DEFAULT}.
64  	 * @return Database assertion mode to use
65  	 */
66  	DatabaseAssertionMode assertionMode() default DatabaseAssertionMode.DEFAULT;
67  
68  	/**
69  	 * Optional table name that can be used to limit the comparison to a specific table.
70  	 * @return the table name
71  	 */
72  	String table() default "";
73  
74  	/**
75  	 * Optional SQL to retrieve the actual subset of the table rows from the database. NOTE: a {@link #table() table
76  	 * name} must also be specified when using a query.
77  	 * @return the SQL Query
78  	 */
79  	String query() default "";
80  
81  	/**
82  	 * If this expectation overrides any others that have been defined at a higher level. Defaults to {@code true}
83  	 * @return if this annotation overrides any others
84  	 */
85  	boolean override() default true;
86  
87  	/**
88  	 * A set of {@link DataSetModifier} that will be applied to the {@link IDataSet} before it is used. Can refer to a
89  	 * static or inner class of the test.
90  	 * @return the modifiers to apply
91  	 */
92  	Class<? extends DataSetModifier>[] modifiers() default {};
93  
94  }
