17  package com.github.springtestdbunit.assertion;
18  
19  import com.github.springtestdbunit.annotation.ExpectedDatabase;
20  
21  /**
22   * Database assertion modes which determine {@link ExpectedDatabase} behavior.
23   *
24   * @author Mario Zagar
25   * @author Mehmet Aslan
26   */
27  public enum DatabaseAssertionMode {
28  
29  	/**
30  	 * Will use default DbUnit data sets assertions.
31  	 */
32  	DEFAULT(new DefaultDatabaseAssertion()),
33  
34  	/**
35  	 * Allows specifying only specific columns and tables in expected data set. Unspecified tables and columns are
36  	 * ignored.
37  	 * <p>
38  	 * <strong>Notes:</strong>
39  	 * <ul>
40  	 * <li>Expected row order must match order in actual data set.</li>
41  	 * <li>Specified columns must match in all rows, e.g. specifying 'column1' value without 'column2' value in one row
42  	 * and only 'column2' value in another is not allowed - both 'column1' and 'column2' values must be specified in all
43  	 * rows.</li>
44  	 * </ul>
45  	 */
46  	NON_STRICT(new NonStrictDatabaseAssertion()),
47  
48  	/**
49  	 * Allows specifying only specific columns and tables in expected data set and ignoring row orders in expected and
50  	 * actual data set. Unspecified tables and columns are ignored. Row orders in expected and actual data sets are
51  	 * ignored.
52  	 * <p>
53  	 * <strong>Notes:</strong>
54  	 * <ul>
55  	 * <li>Expected row order does not need to match order in actual data set.</li>
56  	 * <li>Specified columns must match in all rows, e.g. specifying 'column1' value without 'column2' value in one row
57  	 * and only 'column2' value in another is not allowed - both 'column1' and 'column2' values must be specified in all
58  	 * rows.</li>
59  	 * </ul>
60  	 */
61  	NON_STRICT_UNORDERED(new NonStrictUnorderedDatabaseAssertion());
62  
63  	private DatabaseAssertion databaseAssertion;
64  
65  	private DatabaseAssertionMode(DatabaseAssertion databaseAssertion) {
66  		this.databaseAssertion = databaseAssertion;
67  	}
68  
69  	public DatabaseAssertion getDatabaseAssertion() {
70  		return this.databaseAssertion;
71  	}
72  
73  }
