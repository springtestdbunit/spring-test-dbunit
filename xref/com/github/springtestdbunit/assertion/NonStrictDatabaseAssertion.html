17  package com.github.springtestdbunit.assertion;
18  
19  import java.util.LinkedHashSet;
20  import java.util.List;
21  import java.util.Set;
22  
23  import org.dbunit.Assertion;
24  import org.dbunit.DatabaseUnitException;
25  import org.dbunit.dataset.Column;
26  import org.dbunit.dataset.Columns;
27  import org.dbunit.dataset.DataSetException;
28  import org.dbunit.dataset.FilteredTableMetaData;
29  import org.dbunit.dataset.IDataSet;
30  import org.dbunit.dataset.ITable;
31  import org.dbunit.dataset.ITableMetaData;
32  import org.dbunit.dataset.filter.IColumnFilter;
33  
34  /**
35   * Implements non-strict database assertion strategy : compares data sets ignoring all tables and columns which are not
36   * specified in expected data set but possibly exist in actual data set.
37   *
38   * @author Mario Zagar
39   * @author Sunitha Rajarathnam
40   */
41  class NonStrictDatabaseAssertion implements DatabaseAssertion {
42  
43  	public void assertEquals(IDataSet expectedDataSet, IDataSet actualDataSet, List<IColumnFilter> columnFilters)
44  			throws DatabaseUnitException {
45  		for (String tableName : expectedDataSet.getTableNames()) {
46  			ITable expectedTable = expectedDataSet.getTable(tableName);
47  			ITable actualTable = actualDataSet.getTable(tableName);
48  			assertEquals(expectedTable, actualTable, columnFilters);
49  		}
50  	}
51  
52  	public void assertEquals(ITable expectedTable, ITable actualTable, List<IColumnFilter> columnFilters)
53  			throws DatabaseUnitException {
54  		Set<String> ignoredColumns = getColumnsToIgnore(expectedTable.getTableMetaData(),
55  				actualTable.getTableMetaData(), columnFilters);
56  		Assertion.assertEqualsIgnoreCols(expectedTable, actualTable,
57  				ignoredColumns.toArray(new String[ignoredColumns.size()]));
58  	}
59  
60  	private Set<String> getColumnsToIgnore(ITableMetaData expectedMetaData, ITableMetaData actualMetaData,
61  			List<IColumnFilter> columnFilters) throws DataSetException {
62  		if (columnFilters.size() == 0) {
63  			return getColumnsToIgnore(expectedMetaData, actualMetaData);
64  		}
65  		Set<String> ignoredColumns = new LinkedHashSet<String>();
66  		for (IColumnFilter filter : columnFilters) {
67  			FilteredTableMetaData filteredExpectedMetaData = new FilteredTableMetaData(expectedMetaData, filter);
68  			ignoredColumns.addAll(getColumnsToIgnore(filteredExpectedMetaData, actualMetaData));
69  		}
70  		return ignoredColumns;
71  	}
72  
73  	protected Set<String> getColumnsToIgnore(ITableMetaData expectedMetaData, ITableMetaData actualMetaData)
74  			throws DataSetException {
75  		Column[] notSpecifiedInExpected = Columns.getColumnDiff(expectedMetaData, actualMetaData).getActual();
76  		Set<String> result = new LinkedHashSet<String>();
77  		for (Column column : notSpecifiedInExpected) {
78  			result.add(column.getColumnName());
79  		}
80  		return result;
81  	}
82  
83  }
