16  package org.github.springtestdbunit.assertion;
17  
18  import java.util.LinkedList;
19  import java.util.List;
20  
21  import org.dbunit.Assertion;
22  import org.dbunit.DatabaseUnitException;
23  import org.dbunit.dataset.Column;
24  import org.dbunit.dataset.Columns;
25  import org.dbunit.dataset.DataSetException;
26  import org.dbunit.dataset.IDataSet;
27  import org.dbunit.dataset.ITable;
28  import org.dbunit.dataset.ITableMetaData;
29  
30  /**
31   * Implements non-strict database assertion strategy : compares data sets ignoring all tables and columns which are not
32   * specified in expected data set but possibly exist in actual data set.
33   * 
34   * @author Mario Zagar
35   */
36  class NonStrictDatabaseAssertion implements DatabaseAssertion {
37  
38  	public void assertEquals(IDataSet expectedDataSet, IDataSet actualDataSet) throws DatabaseUnitException {
39  		if (expectedDataSet != actualDataSet) {
40  			for (String tableName : expectedDataSet.getTableNames()) {
41  				ITable expected = expectedDataSet.getTable(tableName);
42  				ITable actual = actualDataSet.getTable(tableName);
43  				String[] ignoredColumns = getColumnsToIgnore(expected.getTableMetaData(), actual.getTableMetaData());
44  				Assertion.assertEqualsIgnoreCols(expected, actual, ignoredColumns);
45  			}
46  		}
47  	}
48  
49  	private String[] getColumnsToIgnore(ITableMetaData expectedMetaData, ITableMetaData actualMetaData)
50  			throws DataSetException {
51  		Column[] notSpecifiedInExpected = Columns.getColumnDiff(expectedMetaData, actualMetaData).getActual();
52  		List<String> result = new LinkedList<String>();
53  		for (Column column : notSpecifiedInExpected) {
54  			result.add(column.getColumnName());
55  		}
56  		return result.toArray(new String[result.size()]);
57  	}
58  }
